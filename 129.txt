package org.example.service.impl;

import com.baomidou.dynamic.datasource.annotation.DS;
import lombok.RequiredArgsConstructor;
import org.example.mapper.McInstrMapper;
import org.example.mapper.McMrktMapper;
import org.example.pojo.McInstrPO;
import org.example.pojo.McMrktPO;
import org.example.service.CaRSCOracleQueryService;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Propagation;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CaRSCOracleQueryServiceImpl implements CaRSCOracleQueryService {

    private final McMrktMapper mcMrktMapper;
    private final McInstrMapper mcInstrMapper;

    @Override
    @DS("oracle")
    @Transactional(propagation = Propagation.REQUIRES_NEW, readOnly = true)
    public McMrktPO findMarketId(String market) {
        return mcMrktMapper.findInstrIdByCode(market);
    }

    @Override
    @DS("oracle")
    @Transactional(propagation = Propagation.REQUIRES_NEW, readOnly = true)
    public List<McInstrPO> findInstrIdByCodeList(String instrument) {
        return mcInstrMapper.findInstrIdByCodeList(instrument);
    }
}
