         // 写入 Oracle（带重试机制，与查询侧对齐，防止网络抖动导致批次数据丢失）
        saveToOracleWithRetry("McAcFundTxnRec", batchIndex, () -> saveToOracleMcAcFundTxn(mcAcFundTxnRecPOList));
        saveToOracleWithRetry("McAcFundRec",    batchIndex, () -> saveToOracleMcAcFundRec(mcAcFundRecPOList));
        saveToOracleWithRetry("McFundTpReltn",  batchIndex, () -> saveToOracleMcFundTpReltn(mcFundTpReltnPOList));
        // CashVoucherMapMSSE 是 SQL Server 侧的映射表（@DS("master")），直接写入，不走 Oracle 重试
        cashVoucherMapMSSEMapper.insert(cashVoucherMapMSSEPOList, 1000);


        /**
     * 带重试的 Oracle 写入封装
     * 遇到网络 IO 类异常（Oracle 17002 / BatchUpdateException）最多重试 3 次，
     * 每次等待递增（10s / 20s / 30s），让 HikariCP 有时间重建可用连接后再重试。
     */
    private void saveToOracleWithRetry(String tableName, int batchIndex, Runnable saveAction) {
        int maxRetry = 3;
        for (int retry = 0; retry <= maxRetry; retry++) {
            try {
                saveAction.run();
                return; // 写入成功，直接返回
            } catch (Exception e) {
                boolean isIoError = e.getMessage() != null &&
                        (e.getMessage().contains("IO Error") ||
                         e.getMessage().contains("17002") ||
                         e.getMessage().contains("connection") ||
                         e.getMessage().contains("BatchUpdateException"));
                if (retry < maxRetry && isIoError) {
                    long waitSeconds = 10L * (retry + 1);
                    log.warn("[Oracle写入] 表={} 第{}批 写入失败（第{}/{}次重试），{}秒后重试。原因：{}",
                            tableName, batchIndex, retry + 1, maxRetry, waitSeconds, e.getMessage());
                    try {
                        Thread.sleep(waitSeconds * 1000L);
                    } catch (InterruptedException ie) {
                        Thread.currentThread().interrupt();
                        throw new RuntimeException("Oracle写入重试等待被中断", ie);
                    }
                } else {
                    // 非 IO 类错误或已超过最大重试次数，直接抛出
                    log.error("[Oracle写入] 表={} 第{}批 写入失败，已重试{}次，放弃。",
                            tableName, batchIndex, retry, e);
                    throw e;
                }
            }
        }
    }
