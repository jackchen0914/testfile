    void batchInsert(@Param("list") List<InstrumentVersion> list);

    List<GenericLog> selectInstrumentLogs();
    
    // 根据条件查询特定Instrument的变更日志
    List<GenericLog> selectInstrumentLogsByKeys(@Param("instrument") String instrument, 
                                                 @Param("market") String market);

    List<GenericLog> selectInstrumentLogsByTimeRange(@Param("startTime") LocalDateTime startTime,
                                                      @Param("endTime") LocalDateTime endTime);
}
