WITH AllOperations AS (
    -- INSERT 操作：一条记录
    SELECT 
        SeqNo,
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        'NEW' AS ChangeType,
        LogColumn,
        AfterValue AS FieldValue
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'INSERT'
        AND AfterValue IS NOT NULL
    
    UNION ALL
    
    -- UPDATE 操作：两条记录（OLD 和 NEW）
    SELECT 
        SeqNo,
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        'OLD' AS ChangeType,
        LogColumn,
        BeforeValue AS FieldValue
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'UPDATE'
        AND BeforeValue IS NOT NULL
    
    UNION ALL
    
    SELECT 
        SeqNo,
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        'NEW' AS ChangeType,
        LogColumn,
        AfterValue AS FieldValue
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'UPDATE'
        AND AfterValue IS NOT NULL
)
SELECT 
    SeqNo,
    InstrumentCode,
    Market,
    LogDateTime,
    LogAction,
    ChangeType,
    LogColumn,
    FieldValue
FROM AllOperations
ORDER BY InstrumentCode, Market, LogDateTime DESC, SeqNo DESC, LogColumn

WITH UpdatePairs AS (
    SELECT 
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogColumn,
        BeforeValue AS OldValue,
        AfterValue AS NewValue,
        LogUserName,
        LogHostName,
        ROW_NUMBER() OVER (
            PARTITION BY TableKey1, TableKey2, LogColumn, LogDateTime
            ORDER BY SeqNo
        ) AS rn
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'UPDATE'
        AND (BeforeValue IS NOT NULL OR AfterValue IS NOT NULL)
),
InsertValues AS (
    SELECT 
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogColumn,
        NULL AS OldValue,
        AfterValue AS NewValue,
        LogUserName,
        LogHostName
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'INSERT'
        AND AfterValue IS NOT NULL
)
SELECT 
    InstrumentCode,
    Market,
    LogDateTime,
    LogColumn,
    OldValue,
    NewValue,
    CASE 
        WHEN OldValue IS NULL AND NewValue IS NOT NULL THEN 'INSERT'
        ELSE 'UPDATE'
    END AS OperationType,
    LogUserName,
    LogHostName
FROM (
    SELECT * FROM UpdatePairs WHERE rn = 1
    UNION ALL
    SELECT * FROM InsertValues
) t
ORDER BY InstrumentCode, Market, LogDateTime DESC, LogColumn

WITH InstrumentLogs AS (
    -- 处理 INSERT 操作：生成一条记录（新值）
    SELECT 
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        LogColumn,
        'NEW' AS ValueType,  -- 标识是新值
        AfterValue AS FieldValue,  -- INSERT 取 AfterValue
        SeqNo,
        ROW_NUMBER() OVER (
            PARTITION BY TableKey1, TableKey2, LogColumn, 'NEW'
            ORDER BY LogDateTime DESC, SeqNo DESC
        ) AS rn
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'INSERT'
        AND AfterValue IS NOT NULL
    
    UNION ALL
    
    -- 处理 UPDATE 操作：生成两条记录（旧值和新值）
    -- 旧值记录
    SELECT 
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        LogColumn,
        'OLD' AS ValueType,
        BeforeValue AS FieldValue,
        SeqNo,
        ROW_NUMBER() OVER (
            PARTITION BY TableKey1, TableKey2, LogColumn, 'OLD'
            ORDER BY LogDateTime DESC, SeqNo DESC
        ) AS rn
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'UPDATE'
        AND BeforeValue IS NOT NULL
    
    UNION ALL
    
    -- 新值记录
    SELECT 
        TableKey1 AS InstrumentCode,
        TableKey2 AS Market,
        LogDateTime,
        LogAction,
        LogColumn,
        'NEW' AS ValueType,
        AfterValue AS FieldValue,
        SeqNo,
        ROW_NUMBER() OVER (
            PARTITION BY TableKey1, TableKey2, LogColumn, 'NEW'
            ORDER BY LogDateTime DESC, SeqNo DESC
        ) AS rn
    FROM GenericLog
    WHERE LogTable = 'Instrument' 
        AND LogAction = 'UPDATE'
        AND AfterValue IS NOT NULL
)
SELECT 
    InstrumentCode,
    Market,
    -- 基础字段
    MAX(CASE WHEN LogColumn = 'Instrument' AND ValueType = 'NEW' THEN FieldValue END) AS Instrument_New,
    MAX(CASE WHEN LogColumn = 'Instrument' AND ValueType = 'OLD' THEN FieldValue END) AS Instrument_Old,
    
    MAX(CASE WHEN LogColumn = 'Market' AND ValueType = 'NEW' THEN FieldValue END) AS Market_New,
    MAX(CASE WHEN LogColumn = 'Market' AND ValueType = 'OLD' THEN FieldValue END) AS Market_Old,
    
    MAX(CASE WHEN LogColumn = 'Name' AND ValueType = 'NEW' THEN FieldValue END) AS Name_New,
    MAX(CASE WHEN LogColumn = 'Name' AND ValueType = 'OLD' THEN FieldValue END) AS Name_Old,
    
    MAX(CASE WHEN LogColumn = 'ShortName' AND ValueType = 'NEW' THEN FieldValue END) AS ShortName_New,
    MAX(CASE WHEN LogColumn = 'ShortName' AND ValueType = 'OLD' THEN FieldValue END) AS ShortName_Old,
    
    MAX(CASE WHEN LogColumn = 'CName' AND ValueType = 'NEW' THEN FieldValue END) AS CName_New,
    MAX(CASE WHEN LogColumn = 'CName' AND ValueType = 'OLD' THEN FieldValue END) AS CName_Old,
    
    MAX(CASE WHEN LogColumn = 'CShortName' AND ValueType = 'NEW' THEN FieldValue END) AS CShortName_New,
    MAX(CASE WHEN LogColumn = 'CShortName' AND ValueType = 'OLD' THEN FieldValue END) AS CShortName_Old,
    
    MAX(CASE WHEN LogColumn = 'InstrumentClass' AND ValueType = 'NEW' THEN FieldValue END) AS InstrumentClass_New,
    MAX(CASE WHEN LogColumn = 'InstrumentClass' AND ValueType = 'OLD' THEN FieldValue END) AS InstrumentClass_Old,
    
    MAX(CASE WHEN LogColumn = 'ProdCode' AND ValueType = 'NEW' THEN FieldValue END) AS ProdCode_New,
    MAX(CASE WHEN LogColumn = 'ProdCode' AND ValueType = 'OLD' THEN FieldValue END) AS ProdCode_Old,
    
    MAX(CASE WHEN LogColumn = 'LotSize' AND ValueType = 'NEW' THEN FieldValue END) AS LotSize_New,
    MAX(CASE WHEN LogColumn = 'LotSize' AND ValueType = 'OLD' THEN FieldValue END) AS LotSize_Old,
    
    MAX(CASE WHEN LogColumn = 'Price' AND ValueType = 'NEW' THEN FieldValue END) AS Price_New,
    MAX(CASE WHEN LogColumn = 'Price' AND ValueType = 'OLD' THEN FieldValue END) AS Price_Old,
    
    MAX(CASE WHEN LogColumn = 'TransactionValueLimit' AND ValueType = 'NEW' THEN FieldValue END) AS TransactionValueLimit_New,
    MAX(CASE WHEN LogColumn = 'TransactionValueLimit' AND ValueType = 'OLD' THEN FieldValue END) AS TransactionValueLimit_Old,
    
    MAX(CASE WHEN LogColumn = 'ExpiryDate' AND ValueType = 'NEW' THEN FieldValue END) AS ExpiryDate_New,
    MAX(CASE WHEN LogColumn = 'ExpiryDate' AND ValueType = 'OLD' THEN FieldValue END) AS ExpiryDate_Old,
    
    MAX(CASE WHEN LogColumn = 'Active' AND ValueType = 'NEW' THEN FieldValue END) AS Active_New,
    MAX(CASE WHEN LogColumn = 'Active' AND ValueType = 'OLD' THEN FieldValue END) AS Active_Old,
    
    -- 继续添加其他字段...
    
    -- 元数据
    MAX(LogDateTime) AS OperationDateTime,
    MAX(CASE WHEN LogColumn = 'LogHostName' AND ValueType = 'NEW' THEN FieldValue END) AS LogHostName,
    MAX(CASE WHEN LogColumn = 'LogUserName' AND ValueType = 'NEW' THEN FieldValue END) AS LogUserName
FROM InstrumentLogs
WHERE rn = 1  -- 取每个类型的最新记录
GROUP BY InstrumentCode, Market, LogDateTime
ORDER BY InstrumentCode, Market, LogDateTime DESC
