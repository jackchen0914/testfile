package org.example.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import java.util.concurrent.ThreadPoolExecutor;

/**
 * 数据迁移线程池配置
 * 针对3000万+数据的大批量迁移场景优化
 */
@Configuration
public class MigrationThreadPoolConfig {

    /**
     * 数据迁移专用线程池
     * 线程数根据服务器配置调整：
     * - CPU核心数8核：建议10-16个线程
     * - CPU核心数16核：建议20-32个线程
     */
    @Bean(name = "migrationExecutor")
    public ThreadPoolTaskExecutor migrationExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        
        // 核心线程数：IO密集型任务，设置为CPU核心数的2倍
        executor.setCorePoolSize(16);
        
        // 最大线程数
        executor.setMaxPoolSize(32);
        
        // 队列容量：避免过大导致内存溢出
        executor.setQueueCapacity(100);
        
        // 线程空闲时间（秒）
        executor.setKeepAliveSeconds(60);
        
        // 线程名称前缀
        executor.setThreadNamePrefix("migration-");
        
        // 拒绝策略：由调用线程执行
        executor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy());
        
        // 等待所有任务完成后再关闭线程池
        executor.setWaitForTasksToCompleteOnShutdown(true);
        
        // 等待时间（秒）
        executor.setAwaitTerminationSeconds(300);
        
        executor.initialize();
        return executor;
    }
}

/**

 * 
 * 优化策略：
 * 1. 多线程并行处理：使用线程池分批并行处理数据
 * 2. 分页流式处理：每次只加载固定数量数据，避免OOM
 * 3. 批量插入优化：减少数据库交互次数
 * 4. 异常隔离：单批次失败不影响整体迁移
 * 5. 进度监控：实时统计迁移进度
 */

    private static final int BATCH_SIZE = 2000;
    
    // 每次批量插入的数据量（Oracle建议1000-2000）
    private static final int INSERT_BATCH_SIZE = 1000;

    @Override
    public String writeProcessedData() {
        long startTime = System.currentTimeMillis();
        log.info("========== 开始数据迁移 ==========");
        
        try {
            // 1. 获取总数据量
            int totalCount = getTotalCount();
            log.info("总数据量: {} 条", totalCount);
            
            if (totalCount == 0) {
                return "没有需要迁移的数据";
            }
            
            // 2. 计算需要的批次数
            int totalBatches = (int) Math.ceil((double) totalCount / BATCH_SIZE);
            log.info("分批处理，每批 {} 条，共 {} 批", BATCH_SIZE, totalBatches);
            
            // 3. 统计计数器
            AtomicInteger successCount = new AtomicInteger(0);
            AtomicInteger failCount = new AtomicInteger(0);
            AtomicLong processedCount = new AtomicLong(0);
            
            // 4. 使用CountDownLatch等待所有任务完成
            CountDownLatch latch = new CountDownLatch(totalBatches);
            
            // 5. 提交所有批次任务到线程池
            for (int i = 0; i < totalBatches; i++) {
                final int batchIndex = i;
                final int offset = i * BATCH_SIZE;
                
                migrationExecutor.execute(() -> {
                    try {
                        // 处理单个批次
                        int processed = processBatch(offset, BATCH_SIZE, batchIndex + 1, totalBatches);
                        successCount.addAndGet(processed);
                        processedCount.addAndGet(processed);
                        
                        // 每10批输出一次进度
                        if ((batchIndex + 1) % 10 == 0) {
                            double progress = (double) processedCount.get() / totalCount * 100;
                            log.info("迁移进度: {}/{} ({:.2f}%)", processedCount.get(), totalCount, progress);
                        }
                    } catch (Exception e) {
                        failCount.incrementAndGet();
                        log.error("批次 {} 处理失败，offset={}", batchIndex + 1, offset, e);
                    } finally {
                        latch.countDown();
                    }
                });
            }
            
            // 6. 等待所有任务完成
            latch.await();
            
            // 7. 输出统计信息
            long endTime = System.currentTimeMillis();
            long duration = (endTime - startTime) / 1000;
            log.info("========== 数据迁移完成 ==========");
            log.info("总耗时: {} 秒 ({} 分钟)", duration, duration / 60);
            log.info("成功: {} 条", successCount.get());
            log.info("失败: {} 批次", failCount.get());
            log.info("平均速度: {} 条/秒", successCount.get() / Math.max(duration, 1));
            
            return String.format("迁移完成！成功: %d 条，失败: %d 批次，耗时: %d 秒", 
                    successCount.get(), failCount.get(), duration);
            
        } catch (Exception e) {
            log.error("数据迁移异常", e);
            return "迁移失败: " + e.getMessage();
        }
    }

    /**
     * 处理单个批次的数据
     */
    private int processBatch(int offset, int limit, int batchIndex, int totalBatches) {
        long batchStartTime = System.currentTimeMillis();
        
        try {
            // 1. 查询当前批次数据
            List<CashVoucherWithRequestDTO> dataList = cashVoucherMapper.selectCashVoucherWithRequest(offset, limit);
            
            if (dataList == null || dataList.isEmpty()) {
                log.warn("批次 {}/{} 没有数据，offset={}", batchIndex, totalBatches, offset);
                return 0;
            }
            
            int actualSize = dataList.size();
            log.info("批次 {}/{} 开始处理，数据量: {}", batchIndex, totalBatches, actualSize);
            
            // 2. 数据转换
            List<McAcFundRecPO> mcAcFundRecPOList = new ArrayList<>(actualSize);
            List<McAcFundTxnRecPO> mcAcFundTxnRecPOList = new ArrayList<>(actualSize);
            List<McFundTpReltnPO> mcFundTpReltnPOList = new ArrayList<>(actualSize);
            List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList = new ArrayList<>(actualSize);
            
            for (CashVoucherWithRequestDTO item : dataList) {
                try {
                    convertAndAddData(item, mcAcFundRecPOList, mcAcFundTxnRecPOList, 
                            mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
                } catch (Exception e) {
                    log.error("数据转换失败，VoucherNo: {}, Market: {}", 
                            item.getVoucherNo(), item.getMarket(), e);
                    // 单条数据转换失败不影响整批
                }
            }
            
            // 3. 批量插入数据（分小批次插入，避免单次插入数据量过大）
            batchInsertData(cashVoucherMapMSSEPOList, mcAcFundTxnRecPOList, 
                    mcAcFundRecPOList, mcFundTpReltnPOList);
            
            long batchDuration = System.currentTimeMillis() - batchStartTime;
            log.info("批次 {}/{} 处理完成，耗时: {} ms，速度: {} 条/秒", 
                    batchIndex, totalBatches, batchDuration, 
                    actualSize * 1000 / Math.max(batchDuration, 1));
            
            return actualSize;
            
        } catch (Exception e) {
            log.error("批次 {}/{} 处理异常，offset={}", batchIndex, totalBatches, offset, e);
            throw new RuntimeException("批次处理失败", e);
        }
    }

    /**
     * 分小批次批量插入数据
     */
    private void batchInsertData(List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList,
                                 List<McAcFundTxnRecPO> mcAcFundTxnRecPOList,
                                 List<McAcFundRecPO> mcAcFundRecPOList,

        // 分小批次插入，避免单次插入数据量过大
        insertInSmallBatches(cashVoucherMapMSSEPOList, this::saveCashVoucherMapMSSE);
        insertInSmallBatches(mcAcFundTxnRecPOList, this::saveToOracleMcAcFundTxn);
        insertInSmallBatches(mcAcFundRecPOList, this::saveToOracleMcAcFundRec);
        insertInSmallBatches(mcFundTpReltnPOList, this::saveToOracleMcFundTpReltn);
    }

    /**
     * 分小批次插入通用方法
     */
    private <T> void insertInSmallBatches(List<T> dataList, BatchInsertFunction<T> insertFunction) {
        if (dataList == null || dataList.isEmpty()) {
            return;
        }
        
        int totalSize = dataList.size();
        for (int i = 0; i < totalSize; i += INSERT_BATCH_SIZE) {
            int end = Math.min(i + INSERT_BATCH_SIZE, totalSize);
            List<T> subList = dataList.subList(i, end);
            insertFunction.insert(subList);
        }
    }

    @FunctionalInterface
    private interface BatchInsertFunction<T> {
        void insert(List<T> list);
    }

    private String isChqAndIsTrnfrTrans(CashVoucherWithRequestDTO items, String type) {
        if (type.equals("chq")) {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().contains("CHQ") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().contains("CHQ") ? "Y" : "N";
            } else return "N";
        } else {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().equals("3. INT TRF") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().equals("3. INT TRF") ? "Y" : "N";
            } else return "N";
        }
    }

    private String isRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (!items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private String isTodayRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private Long txnTypIdValueConvert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        return transactionTypesPO.getSignIndicator().equals("C") ? 1L : 6L;
    }

    private BigDecimal baseCcyEquAmtValueProcess(CashVoucherWithRequestDTO po) {
        ForexRatePO forexRatePO = forexRateMapper.selectRateByDate(po.getCcy(), 
                String.valueOf(po.getConfirmationDate()).split("T")[0]);
        return po.getAmount().multiply(forexRatePO.getXRate()).setScale(8, RoundingMode.HALF_UP);
    }

    private String txnTypActnCdeValueCovert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        if (transactionTypesPO.getSignIndicator() != null) {
            return transactionTypesPO.getSignIndicator().equals("C") ? "IN" : "OUT";
        }
        return "";
    }

    @Resource(name = "migrationExecutor")
    private ThreadPoolTaskExecutor migrationExecutor;

    private static final int BATCH_SIZE = 2000;
    private static final int INSERT_BATCH_SIZE = 1000;
    
    // 最大并发批次数（控制内存使用）
    private static final int MAX_CONCURRENT_BATCHES = 20;

    @Override
    public String writeProcessedData() {
        long startTime = System.currentTimeMillis();
        log.info("========== 开始数据迁移（高级版本） ==========");
        
        try {
            int totalCount = cashVoucherMapper.selectCashVoucherCount();
            log.info("总数据量: {} 条", totalCount);
            
            if (totalCount == 0) {
                return "没有需要迁移的数据";
            }
            
            int totalBatches = (int) Math.ceil((double) totalCount / BATCH_SIZE);
            log.info("分批处理，每批 {} 条，共 {} 批", BATCH_SIZE, totalBatches);
            
            AtomicInteger successCount = new AtomicInteger(0);
            AtomicInteger failCount = new AtomicInteger(0);
            AtomicLong processedCount = new AtomicLong(0);
            
            // 使用CompletableFuture实现流式并行处理
            List<CompletableFuture<Integer>> futures = new ArrayList<>();
            
            for (int i = 0; i < totalBatches; i++) {
                final int batchIndex = i;
                final int offset = i * BATCH_SIZE;
                
                // 创建异步任务
                CompletableFuture<Integer> future = CompletableFuture.supplyAsync(() -> {
                    try {
                        return processBatchAsync(offset, BATCH_SIZE, batchIndex + 1, totalBatches);
                    } catch (Exception e) {
                        log.error("批次 {} 处理失败", batchIndex + 1, e);
                        failCount.incrementAndGet();
                        return 0;
                    }
                }, migrationExecutor).thenApply(processed -> {
                    successCount.addAndGet(processed);
                    long current = processedCount.addAndGet(processed);
                    
                    if ((batchIndex + 1) % 10 == 0) {
                        double progress = (double) current / totalCount * 100;
                        log.info("迁移进度: {}/{} ({:.2f}%)", current, totalCount, progress);
                    }
                    return processed;
                });
                
                futures.add(future);
                
                // 控制并发数，避免内存溢出
                if (futures.size() >= MAX_CONCURRENT_BATCHES) {
                    CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();
                    futures.clear();
                }
            }
            
            // 等待剩余任务完成
            if (!futures.isEmpty()) {
                CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();
            }
            
            long endTime = System.currentTimeMillis();
            long duration = (endTime - startTime) / 1000;
            log.info("========== 数据迁移完成 ==========");
            log.info("总耗时: {} 秒 ({} 分钟)", duration, duration / 60);
            log.info("成功: {} 条", successCount.get());
            log.info("失败: {} 批次", failCount.get());
            log.info("平均速度: {} 条/秒", successCount.get() / Math.max(duration, 1));
            
            return String.format("迁移完成！成功: %d 条，失败: %d 批次，耗时: %d 秒", 
                    successCount.get(), failCount.get(), duration);
            
        } catch (Exception e) {
            log.error("数据迁移异常", e);
            return "迁移失败: " + e.getMessage();
        }
    }

    /**
     * 异步处理单个批次
     */
    private int processBatchAsync(int offset, int limit, int batchIndex, int totalBatches) {
        long batchStartTime = System.currentTimeMillis();
        
        try {
            // 异步查询数据
            List<CashVoucherWithRequestDTO> dataList = cashVoucherMapper.selectCashVoucherWithRequest(offset, limit);
            
            if (dataList == null || dataList.isEmpty()) {
                return 0;
            }
            
            int actualSize = dataList.size();
            
            // 并行转换数据
            List<McAcFundRecPO> mcAcFundRecPOList = new ArrayList<>(actualSize);
            List<McAcFundTxnRecPO> mcAcFundTxnRecPOList = new ArrayList<>(actualSize);
            List<McFundTpReltnPO> mcFundTpReltnPOList = new ArrayList<>(actualSize);
            List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList = new ArrayList<>(actualSize);
            
            // 使用并行流加速数据转换（适用于CPU密集型转换逻辑）
            dataList.parallelStream().forEach(item -> {
                try {
                    synchronized (mcAcFundRecPOList) {
                        convertAndAddData(item, mcAcFundRecPOList, mcAcFundTxnRecPOList, 
                                mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
                    }
                } catch (Exception e) {
                    log.error("数据转换失败，VoucherNo: {}", item.getVoucherNo(), e);
                }
            });
            
            // 异步批量插入（4个表并行插入）
            CompletableFuture<Void> f1 = CompletableFuture.runAsync(() -> 
                    insertInSmallBatches(cashVoucherMapMSSEPOList, this::saveCashVoucherMapMSSE), migrationExecutor);
            
            CompletableFuture<Void> f2 = CompletableFuture.runAsync(() -> 
                    insertInSmallBatches(mcAcFundTxnRecPOList, this::saveToOracleMcAcFundTxn), migrationExecutor);
            
            CompletableFuture<Void> f3 = CompletableFuture.runAsync(() -> 
                    insertInSmallBatches(mcAcFundRecPOList, this::saveToOracleMcAcFundRec), migrationExecutor);
            
            CompletableFuture<Void> f4 = CompletableFuture.runAsync(() -> 
                    insertInSmallBatches(mcFundTpReltnPOList, this::saveToOracleMcFundTpReltn), migrationExecutor);
            
            // 等待所有插入完成
            CompletableFuture.allOf(f1, f2, f3, f4).join();
            
            long batchDuration = System.currentTimeMillis() - batchStartTime;
            log.info("批次 {}/{} 完成，耗时: {} ms，速度: {} 条/秒", 
                    batchIndex, totalBatches, batchDuration, 
                    actualSize * 1000 / Math.max(batchDuration, 1));
            
            return actualSize;
            
        } catch (Exception e) {
            log.error("批次 {} 处理异常", batchIndex, e);
            throw new RuntimeException("批次处理失败", e);
        }
    }

    private <T> void insertInSmallBatches(List<T> dataList, BatchInsertFunction<T> insertFunction) {
        if (dataList == null || dataList.isEmpty()) {
            return;
        }
        
        int totalSize = dataList.size();
        for (int i = 0; i < totalSize; i += INSERT_BATCH_SIZE) {
            int end = Math.min(i + INSERT_BATCH_SIZE, totalSize);
            List<T> subList = dataList.subList(i, end);
            insertFunction.insert(subList);
        }
    }

    @FunctionalInterface
    private interface BatchInsertFunction<T> {
        void insert(List<T> list);
    }

    // ========== 业务逻辑方法 ==========

    private String isChqAndIsTrnfrTrans(CashVoucherWithRequestDTO items, String type) {
        if (type.equals("chq")) {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().contains("CHQ") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().contains("CHQ") ? "Y" : "N";
            } else return "N";
        } else {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().equals("3. INT TRF") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().equals("3. INT TRF") ? "Y" : "N";
            } else return "N";
        }
    }

    private String isRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (!items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private String isTodayRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private Long txnTypIdValueConvert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        return transactionTypesPO.getSignIndicator().equals("C") ? 1L : 6L;
    }

    private BigDecimal baseCcyEquAmtValueProcess(CashVoucherWithRequestDTO po) {
        ForexRatePO forexRatePO = forexRateMapper.selectRateByDate(po.getCcy(), 
                String.valueOf(po.getConfirmationDate()).split("T")[0]);
        return po.getAmount().multiply(forexRatePO.getXRate()).setScale(8, RoundingMode.HALF_UP);
    }

    private String txnTypActnCdeValueCovert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        if (transactionTypesPO.getSignIndicator() != null) {
            return transactionTypesPO.getSignIndicator().equals("C") ? "IN" : "OUT";
        }
        return "";
    }
