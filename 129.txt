package org.example.service.impl;

import com.baomidou.dynamic.datasource.toolkit.DynamicDataSourceContextHolder;
import lombok.RequiredArgsConstructor;
import org.example.mapper.DivAnnCombineMapMSSEMapper;
import org.example.mapper.DivAnnRightsMapMSSEMapper;
import org.example.mapper.DivAnnSplitMapMSSEMapper;
import org.example.pojo.DivAnnCombineMapMSSEPO;
import org.example.pojo.DivAnnRightsMapMSSEPO;
import org.example.pojo.DivAnnSplitMapMSSEPO;
import org.example.service.CaRSCMasterDataService;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CaRSCMasterDataServiceImpl implements CaRSCMasterDataService {

    private final DivAnnRightsMapMSSEMapper divAnnRightsMapMSSEMapper;
    private final DivAnnSplitMapMSSEMapper divAnnSplitMapMSSEMapper;
    private final DivAnnCombineMapMSSEMapper divAnnCombineMapMSSEMapper;

    @Override
    public void insertDivAnnRightsMap(List<DivAnnRightsMapMSSEPO> list) {
        if (list != null && !list.isEmpty()) {
            try {
                DynamicDataSourceContextHolder.push("master");
                divAnnRightsMapMSSEMapper.insert(list);
            } finally {
                DynamicDataSourceContextHolder.poll();
            }
        }
    }

    @Override
    public void insertDivAnnSplitMap(List<DivAnnSplitMapMSSEPO> list) {
        if (list != null && !list.isEmpty()) {
            try {
                DynamicDataSourceContextHolder.push("master");
                divAnnSplitMapMSSEMapper.insert(list);
            } finally {
                DynamicDataSourceContextHolder.poll();
            }
        }
    }

    @Override
    public void insertDivAnnCombine(List<DivAnnCombineMapMSSEPO> list) {
        if (list != null && !list.isEmpty()) {
            try {
                DynamicDataSourceContextHolder.push("master");
                divAnnCombineMapMSSEMapper.insert(list);
            } finally {
                DynamicDataSourceContextHolder.poll();
            }
        }
    }
}




package org.example.service.impl;

import com.baomidou.dynamic.datasource.toolkit.DynamicDataSourceContextHolder;
import lombok.RequiredArgsConstructor;
import org.example.mapper.McInstrMapper;
import org.example.mapper.McMrktMapper;
import org.example.pojo.McInstrPO;
import org.example.pojo.McMrktPO;
import org.example.service.CaRSCOracleQueryService;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CaRSCOracleQueryServiceImpl implements CaRSCOracleQueryService {

    private final McMrktMapper mcMrktMapper;
    private final McInstrMapper mcInstrMapper;

    @Override
    public McMrktPO findMarketId(String market) {
        try {
            DynamicDataSourceContextHolder.push("oracle");
            return mcMrktMapper.findInstrIdByCode(market);
        } finally {
            DynamicDataSourceContextHolder.poll();
        }
    }

    @Override
    public List<McInstrPO> findInstrIdByCodeList(String instrument) {
        try {
            DynamicDataSourceContextHolder.push("oracle");
            return mcInstrMapper.findInstrIdByCodeList(instrument);
        } finally {
            DynamicDataSourceContextHolder.poll();
        }
    }
}
