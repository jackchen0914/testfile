            String relationshipCode = (Objects.equals(item.getRelationship(), "") || item.getRelationship() == null) 
                    ? String.valueOf(System.nanoTime() % 1000000) 
                    : item.getRelationship();
            mcFundTpReltnPO.setFundTpReltnCde(relationshipCode);
            mcFundTpReltnPO.setFundTpReltnDscr(relationshipCode);
            mcFundTpReltnPO.setFundTpReltnPri((long) (18 + System.nanoTime() % 1000));

            batchDeleteLong(txnRidList, ids -> mcAcFundTxnRecMapper.deleteByIds(ids));
            batchDeleteLong(fundRidList, ids -> mcAcFundRecMapper.deleteByIds(ids));
            batchDeleteString(tpReltnCdeList, ids -> mcFundTpReltnMapper.deleteByIds(ids));

    /** Oracle IN 最多 1000 个值，对 Long 类型 ID 列表分批执行 delete */
    private void batchDeleteLong(List<Long> ids, java.util.function.Consumer<List<Long>> deleteFunc) {
        for (int i = 0; i < ids.size(); i += 1000) {
            deleteFunc.accept(ids.subList(i, Math.min(i + 1000, ids.size())));
        }
    }

    /** Oracle IN 最多 1000 个值，对 String 类型 ID 列表分批执行 delete */
    private void batchDeleteString(List<String> ids, java.util.function.Consumer<List<String>> deleteFunc) {
        for (int i = 0; i < ids.size(); i += 1000) {
            deleteFunc.accept(ids.subList(i, Math.min(i + 1000, ids.size())));
        }
    }
