    public String writeProcessedDataByCursorPaging() {
        long startTime = System.currentTimeMillis();
        initCaches();

        long totalCount = cashVoucherMapper.countCashVoucherWithRequest();
        log.info("开始迁移，总数据量: {}", totalCount);

        AtomicInteger processedCount = new AtomicInteger(0);
        AtomicLong lastVoucherNo = new AtomicLong(-1);
        AtomicInteger lastMarketIndex = new AtomicInteger(-1);
        String[] markets = {"HK", "SH", "SZ", "BJ"};

        ExecutorService executor = Executors.newFixedThreadPool(THREAD_POOL_SIZE);
        List<Future<?>> futures = new ArrayList<>();

        while (true) {
            Integer currentLastVoucherNo = lastVoucherNo.get() == -1 ? null : (int) lastVoucherNo.get();
            String currentLastMarket = lastMarketIndex.get() == -1 ? null : markets[lastMarketIndex.get()];

            List<CashVoucherWithRequestDTO> batch = cashVoucherMapper.selectCashVoucherWithRequestByCursor(
                    currentLastVoucherNo, currentLastMarket, CURSOR_PAGE_SIZE);

            if (batch == null || batch.isEmpty()) {
                break;
            }

            CashVoucherWithRequestDTO lastItem = batch.get(batch.size() - 1);
            lastVoucherNo.set(lastItem.getVoucherNo());
            lastMarketIndex.set(Arrays.asList(markets).indexOf(lastItem.getMarket()));

            final List<CashVoucherWithRequestDTO> batchToProcess = batch;
            Future<?> future = executor.submit(() -> {
                processBatch(batchToProcess);
                int processed = processedCount.addAndGet(batchToProcess.size());
                if (processed % 50000 == 0) {
                    log.info("已处理: {}/{} ({}%)", processed, totalCount,
                            String.format("%.2f", processed * 100.0 / totalCount));
                }
            });
            futures.add(future);

            if (futures.size() >= THREAD_POOL_SIZE * 2) {
                waitForSomeFutures(futures);
            }
        }

        for (Future<?> future : futures) {
            try {
                future.get();
            } catch (Exception e) {
                log.error("任务执行异常", e);
            }
        }

        executor.shutdown();
        try {
            executor.awaitTermination(1, TimeUnit.HOURS);
        } catch (InterruptedException e) {
            log.error("线程池关闭被中断", e);
        }

        long endTime = System.currentTimeMillis();
        log.info("迁移完成，总耗时: {}秒，处理数据: {}", (endTime - startTime) / 1000, processedCount.get());
        return "ok";
    }

    public String writeProcessedDataByStream() {
        long startTime = System.currentTimeMillis();
        initCaches();

        long totalCount = cashVoucherMapper.countCashVoucherWithRequest();
        log.info("开始流式迁移，总数据量: {}", totalCount);

        AtomicInteger processedCount = new AtomicInteger(0);
        List<McAcFundRecPO> mcAcFundRecPOList = Collections.synchronizedList(new ArrayList<>(BATCH_SIZE));
        List<McAcFundTxnRecPO> mcAcFundTxnRecPOList = Collections.synchronizedList(new ArrayList<>(BATCH_SIZE));
        List<McFundTpReltnPO> mcFundTpReltnPOList = Collections.synchronizedList(new ArrayList<>(BATCH_SIZE));
        List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList = Collections.synchronizedList(new ArrayList<>(BATCH_SIZE));

        cashVoucherMapper.selectCashVoucherWithRequestStream(context -> {
            CashVoucherWithRequestDTO item = context.getResultObject();
            processSingleItem(item, mcAcFundRecPOList, mcAcFundTxnRecPOList,
                    mcFundTpReltnPOList, cashVoucherMapMSSEPOList);

            int processed = processedCount.incrementAndGet();
            if (processed % BATCH_SIZE == 0) {
                flushToOracle(mcAcFundRecPOList, mcAcFundTxnRecPOList,
                        mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
                log.info("已处理: {}/{} ({}%)", processed, totalCount,
                        String.format("%.2f", processed * 100.0 / totalCount));
            }
        });

        if (!mcAcFundRecPOList.isEmpty()) {
            flushToOracle(mcAcFundRecPOList, mcAcFundTxnRecPOList,
                    mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
        }

        long endTime = System.currentTimeMillis();
        log.info("流式迁移完成，总耗时: {}秒，处理数据: {}", (endTime - startTime) / 1000, processedCount.get());
        return "ok";
    }

    private void initCaches() {
        if (transactionTypesCache == null || forexRateCache == null) {
            synchronized (cacheLock) {
                if (transactionTypesCache == null) {
                    List<TransactionTypesPO> txnTypes = transactionTypesMapper.selectAllTransactionTypes();
                    transactionTypesCache = new ConcurrentHashMap<>();
                    for (TransactionTypesPO po : txnTypes) {
                        String key = po.getStkMoney() + "_" + po.getTransactionTypes();
                        transactionTypesCache.put(key, po);
                    }
                    log.info("TransactionTypes缓存加载完成，共{}条", transactionTypesCache.size());
                }
                if (forexRateCache == null) {
                    List<ForexRatePO> rates = forexRateMapper.selectAllForexRates();
                    forexRateCache = new ConcurrentHashMap<>();
                    for (ForexRatePO po : rates) {
                        if (po.getDate() != null) {
                            String key = po.getCcy() + "_" + po.getDate().toLocalDate().toString();
                            forexRateCache.put(key, po);
                        }
                    }
                    log.info("ForexRate缓存加载完成，共{}条", forexRateCache.size());
                }
            }
        }
    }

    private void waitForSomeFutures(List<Future<?>> futures) {
        Iterator<Future<?>> iterator = futures.iterator();
        while (iterator.hasNext()) {
            Future<?> future = iterator.next();
            if (future.isDone()) {
                try {
                    future.get();
                } catch (Exception e) {
                    log.error("任务执行异常", e);
                }
                iterator.remove();
            }
        }
    }

    private void processBatch(List<CashVoucherWithRequestDTO> batch) {
        List<McAcFundRecPO> mcAcFundRecPOList = new ArrayList<>(batch.size());
        List<McAcFundTxnRecPO> mcAcFundTxnRecPOList = new ArrayList<>(batch.size());
        List<McFundTpReltnPO> mcFundTpReltnPOList = new ArrayList<>(batch.size());
        List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList = new ArrayList<>(batch.size());

        for (int i = 0; i < batch.size(); i++) {
            CashVoucherWithRequestDTO items = batch.get(i);
            processSingleItemWithIndex(items, i, mcAcFundRecPOList, mcAcFundTxnRecPOList,
                    mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
        }

        flushToOracle(mcAcFundRecPOList, mcAcFundTxnRecPOList,
                mcFundTpReltnPOList, cashVoucherMapMSSEPOList);
    }

    private void flushToOracle(List<McAcFundRecPO> mcAcFundRecPOList,
                               List<McAcFundTxnRecPO> mcAcFundTxnRecPOList,
                               List<McFundTpReltnPO> mcFundTpReltnPOList,
                               List<CashVoucherMapMSSEPO> cashVoucherMapMSSEPOList) {
        if (mcAcFundRecPOList.isEmpty()) {
            return;
        }

        List<McAcFundRecPO> recList = new ArrayList<>(mcAcFundRecPOList);
        List<McAcFundTxnRecPO> txnList = new ArrayList<>(mcAcFundTxnRecPOList);
        List<McFundTpReltnPO> reltnList = new ArrayList<>(mcFundTpReltnPOList);
        List<CashVoucherMapMSSEPO> mapList = new ArrayList<>(cashVoucherMapMSSEPOList);

        mcAcFundRecPOList.clear();
        mcAcFundTxnRecPOList.clear();
        mcFundTpReltnPOList.clear();
        cashVoucherMapMSSEPOList.clear();

        saveToOracleInBatches(mapList, recList, txnList, reltnList);
    }

    @Transactional(rollbackFor = Exception.class)
    public void saveToOracleInBatches(List<CashVoucherMapMSSEPO> mapList,
                                      List<McAcFundRecPO> recList,
                                      List<McAcFundTxnRecPO> txnList,
                                      List<McFundTpReltnPO> reltnList) {
        batchInsertWithSize(mapList, cashVoucherMapMSSEMapper::insert);
        batchInsertWithSize(txnList, mcAcFundTxnRecMapper::batchInsert);
        batchInsertWithSize(recList, mcAcFundRecMapper::batchInsert);
        batchInsertWithSize(reltnList, mcFundTpReltnMapper::batchInsert);
    }

    private <T> void batchInsertWithSize(List<T> list, java.util.function.Consumer<List<T>> insertFunction) {
        if (list == null || list.isEmpty()) {
            return;
        }
        for (int i = 0; i < list.size(); i += BATCH_SIZE) {
            int end = Math.min(i + BATCH_SIZE, list.size());
            List<T> subList = list.subList(i, end);
            insertFunction.accept(subList);
        }
    }

    private String isChqAndIsTrnfrTrans(CashVoucherWithRequestDTO items, String type) {
        if ("chq".equals(type)) {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().contains("CHQ") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().contains("CHQ") ? "Y" : "N";
            }
            return "N";
        } else {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().equals("3. INT TRF") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().equals("3. INT TRF") ? "Y" : "N";
            }
            return "N";
        }
    }

    private String isRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (!items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus() != null && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private String isTodayRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (items.getCancelDate().equals(items.getConfirmationDate()) && items.getStatus() != null && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private Long txnTypIdValueConvertFromCache(String code) {
        if (code == null) return 1L;
        String key = "M_" + code;
        TransactionTypesPO po = transactionTypesCache.get(key);
        if (po != null && po.getSignIndicator() != null) {
            return "C".equals(po.getSignIndicator()) ? 1L : 6L;
        }
        return 1L;
    }

    private BigDecimal baseCcyEquAmtValueProcessFromCache(CashVoucherWithRequestDTO po) {
        if (po.getCcy() == null || po.getConfirmationDate() == null || po.getAmount() == null) {
            return BigDecimal.ZERO;
        }
        String key = po.getCcy() + "_" + po.getConfirmationDate().toLocalDate().toString();
        ForexRatePO forexRatePO = forexRateCache.get(key);
        if (forexRatePO != null && forexRatePO.getXRate() != null) {
            return po.getAmount().multiply(forexRatePO.getXRate()).setScale(8, RoundingMode.HALF_UP);
        }
        return BigDecimal.ZERO;
    }

    private String txnTypActnCdeValueCovertFromCache(String code) {
        if (code == null) return "";
        String key = "M_" + code;
        TransactionTypesPO po = transactionTypesCache.get(key);
        if (po != null && po.getSignIndicator() != null) {
            return "C".equals(po.getSignIndicator()) ? "IN" : "OUT";
        }
        return "";
    }
