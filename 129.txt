    @Override
    @Transactional(rollbackFor = Exception.class)
    public void buildInstrumentVersion() {
        log.info("开始构建InstrumentVersion数据...");
        
        // 1. 查询所有Instrument日志
        List<GenericLog> allLogs = instrumentVersionMapper.selectInstrumentLogs();
        if (allLogs.isEmpty()) {
            log.info("未找到Instrument日志记录");
            return;
        }
        
        log.info("共查询到 {} 条Instrument日志记录", allLogs.size());
        
        // 2. 按Instrument+Market+LogDateTime分组
        Map<String, List<GenericLog>> groupedLogs = groupLogsByInstrumentAndTime(allLogs);
        
        // 3. 构建InstrumentVersion记录
        List<InstrumentVersion> versionList = new ArrayList<>();
        
        for (Map.Entry<String, List<GenericLog>> entry : groupedLogs.entrySet()) {
            try {
                InstrumentVersion version = buildVersionFromLogs(entry.getValue());
                if (version != null) {
                    versionList.add(version);
                }
            } catch (Exception e) {
                log.error("处理日志组 {} 时发生错误", entry.getKey(), e);
            }
        }
        
        // 4. 批量插入
        if (!versionList.isEmpty()) {
            instrumentVersionMapper.batchInsert(versionList);
            log.info("成功插入 {} 条InstrumentVersion记录", versionList.size());
        }
    }

    /**
     * 按Instrument、Market和LogDateTime分组
     * 同一时间点的变更归为一组
     */
    private Map<String, List<GenericLog>> groupLogsByInstrumentAndTime(List<GenericLog> logs) {
        return logs.stream()
            .filter(log -> INSTRUMENT_FIELDS.contains(log.getLogColumn()))
            .collect(Collectors.groupingBy(log -> 
                log.getTableKey1() + "|" + 
                log.getTableKey2() + "|" + 
                log.getLogDateTime().toString()
            ));
    }

    /**
     * 从一组日志构建InstrumentVersion对象
     */
    private InstrumentVersion buildVersionFromLogs(List<GenericLog> logs) {
        if (logs == null || logs.isEmpty()) {
            return null;
        }

        InstrumentVersion version = new InstrumentVersion();
        GenericLog firstLog = logs.get(0);
        
        // 设置基础信息
        version.setInstrument(firstLog.getTableKey1());
        version.setMarket(firstLog.getTableKey2());
        version.setLogDateTime(firstLog.getLogDateTime());
        version.setLogHostName(firstLog.getLogHostName());
        version.setLogUserName(firstLog.getLogUserName());
        
        // 按LogAction分类处理
        Map<String, List<GenericLog>> actionGroups = logs.stream()
            .collect(Collectors.groupingBy(GenericLog::getLogAction));
        
        // 处理INSERT记录
        List<GenericLog> insertLogs = actionGroups.get("INSERT");
        if (insertLogs != null && !insertLogs.isEmpty()) {
            processInsertLogs(version, insertLogs);
        }
        
        // 处理UPDATE记录
        List<GenericLog> updateLogs = actionGroups.get("UPDATE");
        if (updateLogs != null && !updateLogs.isEmpty()) {
            processUpdateLogs(version, updateLogs);
        }
        
        return version;
    }

    /**
     * 处理INSERT类型的日志
     */
    private void processInsertLogs(InstrumentVersion version, List<GenericLog> insertLogs) {
        for (GenericLog log : insertLogs) {
            setFieldValue(version, log.getLogColumn(), log.getAfterValue());
        }
    }

    /**
     * 处理UPDATE类型的日志
     * UPDATE日志中，AfterValue是更新后的值
     */
    private void processUpdateLogs(InstrumentVersion version, List<GenericLog> updateLogs) {
        for (GenericLog log : updateLogs) {
            setFieldValue(version, log.getLogColumn(), log.getAfterValue());
        }
    }

    private LocalDateTime parseLocalDateTime(String dateStr) {
        if (!StringUtils.hasText(dateStr)) return null;
        try {
            // 处理 "2025-10-21 00:00:00" 格式
            return LocalDateTime.parse(dateStr.replace(" ", "T"));
        } catch (Exception e) {
            log.warn("日期解析失败: {}", dateStr);
            return null;
        }
    }

    private Date parseDate(String dateStr) {
        if (!StringUtils.hasText(dateStr) || "NULL".equals(dateStr) || "[NULL]".equals(dateStr)) {
            return null;
        }
        try {
            // 处理 "2022-07-08 00:00:00" 格式
            return java.sql.Timestamp.valueOf(dateStr);
        } catch (Exception e) {
            log.warn("日期解析失败: {}", dateStr);
            return null;
        }
    }
