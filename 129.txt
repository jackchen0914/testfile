package org.example.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.concurrent.LinkedBlockingQueue;
import java.util.concurrent.ThreadPoolExecutor;
import java.util.concurrent.TimeUnit;

/**
 * 数据迁移专用线程池配置
 * <p>
 * corePoolSize = 8：按年份最多 8 个并发任务（2018~2026）
 * maximumPoolSize = 8：不允许超出核心数，避免数据库连接池被打爆
 * keepAliveTime = 60s：空闲线程存活时间
 * 队列容量 = 20：防止任务堆积，超出时走 CallerRunsPolicy（调用方线程兜底执行）
 * </p>
 *
 * @author JackChen
 */
@Configuration
public class MigrationThreadPoolConfig {

    /** 核心 / 最大线程数 = 年份区间上限，默认 8 */
    private static final int CORE_POOL_SIZE = 8;

    /** 单批次每页记录数 */
    public static final int PAGE_SIZE = 500;

    @Bean(name = "migrationThreadPool")
    public ThreadPoolExecutor migrationThreadPool() {
        return new ThreadPoolExecutor(
                CORE_POOL_SIZE,
                CORE_POOL_SIZE,
                60L,
                TimeUnit.SECONDS,
                new LinkedBlockingQueue<>(20),
                r -> {
                    Thread t = new Thread(r);
                    t.setName("migration-worker-" + t.getId());
                    t.setDaemon(false);
                    return t;
                },
                // 队列满时由调用方线程直接执行，不丢任务
                new ThreadPoolExecutor.CallerRunsPolicy()
        );
    }
}

/**

 * 1. 先查询 ValueDate 的最小年份和最大年份，确定迁移区间
 * 2. 每一年分配一个 CompletableFuture 任务（最多 8 个并发）
 * 3. 每个任务内部按 PAGE_SIZE 分页循环拉取该年数据并写入 Oracle
 *
 * @author JackChen
 * @since 2025-12-12
 */
@Service

    // 主入口
    // -----------------------------------------------------------------------

    @Override
    public String writeProcessedData() {
        // 1. 查询 ValueDate 最小/最大年份，确定迁移区间
        LocalDateTime minDate = cashVoucherMapper.selectMinValueDate();
        LocalDateTime maxDate = cashVoucherMapper.selectMaxValueDate();

        if (minDate == null || maxDate == null) {
            log.warn("[CashVoucher迁移] ValueDate 为空，无数据需要迁移");
            return "no data";
        }

        int startYear = minDate.getYear();
        int endYear   = maxDate.getYear();
        log.info("[CashVoucher迁移] 迁移区间: {} 年 ~ {} 年，共 {} 年",
                startYear, endYear, endYear - startYear + 1);

        // 2. 每一年构建一个 CompletableFuture 异步任务
        List<CompletableFuture<Void>> futures = new ArrayList<>();
        for (int year = startYear; year <= endYear; year++) {
            final int targetYear = year;
            futures.add(CompletableFuture.runAsync(
                    () -> migrateOneYear(targetYear), migrationThreadPool));
        }

        // 3. 等待所有年份迁移完成
        try {
            CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();
        } catch (Exception e) {
            log.error("[CashVoucher迁移] 多线程迁移过程中发生异常", e);
            return "error: " + e.getMessage();
        }

        log.info("[CashVoucher迁移] 全部年份迁移完成");
        return "ok";
    }

    // -----------------------------------------------------------------------
    // 单年迁移：内部分页循环
    // -----------------------------------------------------------------------

    private void migrateOneYear(int year) {
        long total = cashVoucherMapper.countByYearRange(year, year);
        if (total == 0) {
            log.info("[CashVoucher迁移] {} 年无数据，跳过", year);
            return;
        }

        int  pageSize  = MigrationThreadPoolConfig.PAGE_SIZE;
        long totalPage = (total + pageSize - 1) / pageSize;
        log.info("[CashVoucher迁移] {} 年共 {} 条，分 {} 页（每页 {} 条）",
                year, total, totalPage, pageSize);

        for (int page = 0; page < totalPage; page++) {
            List<CashVoucherWithRequestDTO> batch =
                    cashVoucherMapper.selectByYearRange(year, year, page * pageSize, pageSize);
            if (batch == null || batch.isEmpty()) break;
            log.info("[CashVoucher迁移] {} 年 第 {}/{} 页，本批 {} 条",
                    year, page + 1, totalPage, batch.size());
            processBatch(batch);
        }

        log.info("[CashVoucher迁移] {} 年迁移完成", year);
    }

    private String isChqAndIsTrnfrTrans(CashVoucherWithRequestDTO items, String type) {
        if (type.equals("chq")) {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().contains("CHQ") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().contains("CHQ") ? "Y" : "N";
            } else return "N";
        } else {
            if (!StringUtils.isEmpty(items.getMode())) {
                return items.getMode().equals("3. INT TRF") ? "Y" : "N";
            } else if (!StringUtils.isEmpty(items.getWithdrawMode())) {
                return items.getWithdrawMode().equals("3. INT TRF") ? "Y" : "N";
            } else return "N";
        }
    }

    private String isRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (!items.getCancelDate().equals(items.getConfirmationDate())
                    && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private String isTodayRevTrans(CashVoucherWithRequestDTO items) {
        if (items.getCancelDate() != null && items.getConfirmationDate() != null) {
            if (items.getCancelDate().equals(items.getConfirmationDate())
                    && items.getStatus().contains("CAN")) {
                return "Y";
            }
        }
        return "N";
    }

    private Long txnTypIdValueConvert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        return transactionTypesPO.getSignIndicator().equals("C") ? 1L : 6L;
    }

    private BigDecimal baseCcyEquAmtValueProcess(CashVoucherWithRequestDTO po) {
        ForexRatePO forexRatePO = forexRateMapper.selectRateByDate(
                po.getCcy(), String.valueOf(po.getConfirmationDate()).split("T")[0]);
        return po.getAmount().multiply(forexRatePO.getXRate()).setScale(8, RoundingMode.HALF_UP);
    }

    private String txnTypActnCdeValueCovert(String code) {
        TransactionTypesPO transactionTypesPO = transactionTypesMapper.selectTxnTypeCode(code);
        if (transactionTypesPO.getSignIndicator() != null) {
            return transactionTypesPO.getSignIndicator().equals("C") ? "IN" : "OUT";
        }
        return "";
    }
