    @Override
    public String writeProcessedData(long maxRows) {
        LocalDateTime minDate = instrumentVoucherMapper.selectMinVoucherDate();
        LocalDateTime maxDate = instrumentVoucherMapper.selectMaxVoucherDate();
        if (minDate == null || maxDate == null) {
            log.warn("InstrumentVoucher 表中无有效 VoucherDate 数据，跳过迁移");
            return "no data";
        }
        int startYear = minDate.getYear();
        int endYear   = maxDate.getYear();
        boolean limited = maxRows > 0;
        if (limited) log.info("[测试模式] 最大迁移行数限制：{} 条", maxRows);
        log.info("迁移年份区间：{} ~ {}，共 {} 年", startYear, endYear, (endYear - startYear + 1));

        // ① 全局预加载 TransactionTypes StkMoney='S'（全表很小，只查一次）
        Map<String, TransactionTypesPO> txnActionMap = transactionTypesMapper.selectAllTxnTypeAction()
                .stream().collect(Collectors.toMap(
                        TransactionTypesPO::getTransactionTypes, po -> po, (a, b) -> a));

        // ② 全局预加载 McMrkt（市场代码表全量，只查一次）
        Map<String, Long> mrktCdeToIdMap = mcMrktMapper.findAllMrkt().stream()
                .filter(po -> po.getMrktCde() != null && po.getMrktId() != null)
                .collect(Collectors.toMap(McMrktPO::getMrktCde, McMrktPO::getMrktId, (a, b) -> a));

        log.info("全局预加载完成：TransactionTypes={} 条，McMrkt={} 条",
                txnActionMap.size(), mrktCdeToIdMap.size());

        final AtomicLong remainingRows = new AtomicLong(limited ? maxRows : Long.MAX_VALUE);
        List<CompletableFuture<Void>> futures = new ArrayList<>();
        for (int year = startYear; year <= endYear; year++) {
            final int targetYear = year;
            futures.add(CompletableFuture.runAsync(
                    () -> processOneYear(targetYear, remainingRows, txnActionMap, mrktCdeToIdMap),
                    migrationThreadPool));
        }
        try {
            CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).get();
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            log.error("迁移任务被中断", e);
            return "interrupted";
        } catch (ExecutionException e) {
            log.error("迁移任务执行异常", e.getCause());
            return "error: " + e.getCause().getMessage();
        } finally {
            txnActionMap.clear();
            mrktCdeToIdMap.clear();
        }
        log.info("全部年份迁移完成，区间：{} ~ {}", startYear, endYear);
        return "successfully";
    }

    private void processOneYear(int year, AtomicLong remainingRows,
                                Map<String, TransactionTypesPO> txnActionMap,
                                Map<String, Long> mrktCdeToIdMap) {
        String originalName = Thread.currentThread().getName();
        Thread.currentThread().setName("migration-instr-year-" + year);
        try {
            LocalDateTime yearStart = LocalDateTime.of(year, 1, 1, 0, 0, 0);
            LocalDateTime yearEnd   = LocalDateTime.of(year + 1, 1, 1, 0, 0, 0);
            long totalCount = instrumentVoucherMapper.countByVoucherDateRange(yearStart, yearEnd);
            long yearLimit = Math.min(totalCount, remainingRows.get());
            if (yearLimit <= 0) { log.info("[{}年] 已达到最大迁移行数限制，跳过", year); return; }
            long totalBatches = (long) Math.ceil((double) yearLimit / PAGE_SIZE);
            log.info("[{}年] 总记录数：{}，本次处理上限：{}，预计共 {} 批",
                    year, totalCount, yearLimit, totalBatches);
            int offset = 0, batchIndex = 0;
            long processedInYear = 0;
            while (true) {
                long leftInYear = yearLimit - processedInYear;
                if (leftInYear <= 0) break;
                int fetchSize = (int) Math.min(PAGE_SIZE, leftInYear);
                List<InstrumentVoucherPO> batch = null;
                int maxRetry = 3;
                for (int retry = 0; retry <= maxRetry; retry++) {
                    try {
                        batch = instrumentVoucherMapper.selectByVoucherDateRangePage(
                                yearStart, yearEnd, offset, fetchSize);
                        break;
                    } catch (Exception e) {
                        if (retry < maxRetry) {
                            log.warn("[{}年] 第{}批查询失败({}/{}次重试)，原因：{}",
                                    year, batchIndex + 1, retry + 1, maxRetry, e.getMessage());
                            try { Thread.sleep(10_000L * (retry + 1)); }
                            catch (InterruptedException ie) { Thread.currentThread().interrupt(); throw new RuntimeException(ie); }
                        } else {
                            log.error("[{}年] 第{}批查询重试{}次仍失败", year, batchIndex + 1, maxRetry, e);
                            throw e;
                        }
                    }
                }
                if (batch == null || batch.isEmpty()) break;
                batchIndex++;
                int fetchedSize = batch.size();
                log.info("[{}年] 第{}/{} 批，本批 {} 条，开始处理",
                        year, batchIndex, totalBatches, fetchedSize);
                processBatch(batch, batchIndex, txnActionMap, mrktCdeToIdMap);
                processedInYear += fetchedSize;
                remainingRows.addAndGet(-fetchedSize);
                batch.clear();
                if (fetchedSize < fetchSize) break;
                offset += fetchSize;
            }
            log.info("[{}年] 迁移完成，本年共处理 {} 条", year, processedInYear);
        } catch (Exception e) {
            log.error("[{}年] 迁移异常", year, e);
        } finally {
            Thread.currentThread().setName(originalName);
        }
    }

    /**
     * 批次内预加载 McInstr（按本批去重后的 instrument 列表批量查）
     * key = instr_dsply_cde
     */
    private Map<String, McInstrPO> buildInstrMap(List<InstrumentVoucherPO> batch) {
        List<String> codes = batch.stream()
                .map(InstrumentVoucherPO::getInstrument)
                .filter(Objects::nonNull)
                .distinct()
                .collect(Collectors.toList());
        if (codes.isEmpty()) return Collections.emptyMap();
        return mcInstrMapper.findInstrListByCodes(codes).stream()
                .filter(po -> po.getInstrDsplyCde() != null)
                .collect(Collectors.toMap(McInstrPO::getInstrDsplyCde, po -> po, (a, b) -> a));
    }

    /**
     * 批次内预加载 McInstclVer
     * 先收集本批所有 instclId，批量查所有版本记录，
     * 再按 "instclId_dateStr" 匹配有效区间，构建查找 Map
     */
    private Map<String, Long> buildInstclVidMap(List<InstrumentVoucherPO> batch,
                                                Map<String, McInstrPO> instrMap) {
        List<Long> instclIds = batch.stream()
                .map(po -> {
                    McInstrPO instr = instrMap.get(po.getInstrument());
                    return instr != null ? instr.getInstclId() : null;
                })
                .filter(Objects::nonNull)
                .distinct()
                .collect(Collectors.toList());
        if (instclIds.isEmpty()) return Collections.emptyMap();

        List<McInstclVerPO> verList = mcInstclVerMapper.findInstclVerListByInstclIds(instclIds);
        // key = "instclId_yyyy-MM-dd"
        Map<String, Long> result = new HashMap<>();
        for (InstrumentVoucherPO po : batch) {
            if (po.getConfirmationDate() == null) continue;
            McInstrPO instr = instrMap.get(po.getInstrument());
            if (instr == null || instr.getInstclId() == null) continue;
            Long instclId = instr.getInstclId();
            String dateKey = instclId + "_" + po.getConfirmationDate().toLocalDate().toString();
            if (result.containsKey(dateKey)) continue;
            LocalDate confDate = po.getConfirmationDate().toLocalDate();
            for (McInstclVerPO ver : verList) {
                if (!instclId.equals(ver.getInstclId())) continue;
                if (ver.getVerEffDate() == null || ver.getVerExprDate() == null) continue;
                LocalDate eff  = ver.getVerEffDate().toLocalDate();
                LocalDate expr = ver.getVerExprDate().toLocalDate();
                if (!confDate.isBefore(eff) && !confDate.isAfter(expr)) {
                    result.put(dateKey, ver.getInstclVid());
                    break;
                }
            }
        }
        return result;
    }

            mcLoctnStkTxnDtlPO.setStkStatCde(stkStatCde);
            mcLoctnStkTxnDtlPO.setRecVerNum(0L);
            mcLoctnStkTxnDtlPO.setInitTime(LocalDateTime.now());
            mcLoctnStkTxnDtlPO.setLastUpdTime(LocalDateTime.now());
            mcLoctnStkTxnDtlPO.setLastUpdBy("MIG");
            mcLoctnStkTxnDtlPO.setTagSeq(0L);

            InstrumentVoucherMapMSSEPO instrumentVoucherMapMSSEPO = new InstrumentVoucherMapMSSEPO();
            instrumentVoucherMapMSSEPO.setVoucherNo(items.getVoucherNo());
            instrumentVoucherMapMSSEPO.setMarket(items.getMarket());
            instrumentVoucherMapMSSEPO.setAcStkTxnDtlStkStatId(acStkTxnDtlStkStatId);
            instrumentVoucherMapMSSEPO.setAcStkTxnDtlId(acStkTxnDtlRelevanceId);
            instrumentVoucherMapMSSEPO.setStkMemoTxnId(setStkMemoTxnRelevanceId);
            instrumentVoucherMapMSSEPO.setStkMemoTxnReqId(stkMemoTxnReqId);

            mcAcStkTxnDtlStatPOList.add(mcAcStkTxnDtlStatPO);
            mcStkTxnPOList.add(mcStkTxnPO);
            mcAcStkTxnDtlPOList.add(mcAcStkTxnDtlPO);
            mcStkMemoTxnPOList.add(mcStkMemoTxnPO);
            mcStkMemoTxnReqPOList.add(mcStkMemoTxnReqPO);
            mcLoctnStkTxnDtlPOList.add(mcLoctnStkTxnDtlPO);
            instrumentVoucherMapMSSEPOList.add(instrumentVoucherMapMSSEPO);
        }

        // 写入 Oracle（带重试机制）
        saveToOracleWithRetry("McStkTxn",          batchIndex, () -> saveToMcStkTxn(mcStkTxnPOList));
        saveToOracleWithRetry("McAcStkTxnDtl",     batchIndex, () -> saveToMcAcStkTxnDtl(mcAcStkTxnDtlPOList));
        saveToOracleWithRetry("McAcStkTxnDtlStat", batchIndex, () -> saveToMcAcStkTxnDtlStat(mcAcStkTxnDtlStatPOList));
        saveToOracleWithRetry("McStkMemoTxn",      batchIndex, () -> saveToMcStkMemoTxn(mcStkMemoTxnPOList));
        saveToOracleWithRetry("McStkMemoTxnReq",   batchIndex, () -> saveToMcStkMemoTxnReq(mcStkMemoTxnReqPOList));
        saveToOracleWithRetry("McLoctnStkTxnDtl",  batchIndex, () -> saveToMcLoctnStkTxnDtl(mcLoctnStkTxnDtlPOList));
        // InstrumentVoucherMapMSSE 是 SQL Server 侧的映射表，直接写入
        instrumentVoucherMapMSSEMapper.insert(instrumentVoucherMapMSSEPOList);

        mcAcStkTxnDtlStatPOList.clear();
        mcAcStkTxnDtlPOList.clear();
        mcStkTxnPOList.clear();
        mcStkMemoTxnPOList.clear();
        mcStkMemoTxnReqPOList.clear();
        mcLoctnStkTxnDtlPOList.clear();
        instrumentVoucherMapMSSEPOList.clear();
    }

    private String stkStatCdeTrans(String location, String txnType,
                                   Map<String, TransactionTypesPO> txnActionMap) {
        switch (location == null ? "" : location) {
            case "NOM":   return "NOMINEE";
            case "OWM":   return "OWNERNAME";
            case "BONUS": return "BONUS";
            case "ST":

        }

    /**
     * 带重试的 Oracle 写入封装
     */
    private void saveToOracleWithRetry(String tableName, int batchIndex, Runnable saveAction) {
        int maxRetry = 3;
        for (int retry = 0; retry <= maxRetry; retry++) {
            try {
                saveAction.run();
                return;
            } catch (Exception e) {
                boolean isIoError = e.getMessage() != null &&
                        (e.getMessage().contains("IO Error") ||
                         e.getMessage().contains("17002") ||
                         e.getMessage().contains("connection") ||
                         e.getMessage().contains("BatchUpdateException"));
                if (retry < maxRetry && isIoError) {
                    long waitSeconds = 10L * (retry + 1);
                    log.warn("[Oracle写入] 表={} 第{}批 写入失败（第{}/{}次重试），{}秒后重试。原因：{}",
                            tableName, batchIndex, retry + 1, maxRetry, waitSeconds, e.getMessage());
                    try { Thread.sleep(waitSeconds * 1000L); }
                    catch (InterruptedException ie) { Thread.currentThread().interrupt(); throw new RuntimeException("Oracle写入重试等待被中断", ie); }
                } else {
                    log.error("[Oracle写入] 表={} 第{}批 写入失败，已重试{}次，放弃。", tableName, batchIndex, retry, e);
                    throw e;
                }
            }
        }
    }

        //2019-06-05 00:00:00.000
        if (po.getCancelDate() != null && po.getConfirmationDate() != null) {
            if (!po.getCancelDate().equals(po.getConfirmationDate()) && po.getStatus().contains("CAN")) return "Y";
        }
        return "N";
    }

        //2019-06-05 00:00:00.000
        if (po.getCancelDate() != null && po.getConfirmationDate() != null) {
            if (po.getCancelDate().equals(po.getConfirmationDate()) && po.getStatus().contains("CAN")) return "Y";
        }
        return "N";
    }
