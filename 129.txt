    private BigDecimal findBaseCcyEquAmtValue(CashTransferAllDTO dto) {
        ForexRatePO ratePO = forexRateMapper.selectRateByDate(dto.getCCYFrom(), String.valueOf(dto.getTransferDate()).replaceAll("T"," "));
        if(ratePO!=null) {//BigDecimal.ONE.divide(ratePO.getXRate(), 4, RoundingMode.HALF_UP)
            return ratePO.getXRate() != null ? dto.getAmountFrom().multiply(ratePO.getXRate()).setScale(2, RoundingMode.HALF_UP) : BigDecimal.ZERO;
        }
        return BigDecimal.ZERO;
    }

    private BigDecimal findBaseCcyEquToAmtValue(CashTransferAllDTO dto) {
        ForexRatePO ratePO = forexRateMapper.selectRateByDate(dto.getCCYFrom(), String.valueOf(dto.getTransferDate()).replaceAll("T"," "));
        if(ratePO!=null) {
            return ratePO.getXRate() != null ? ratePO.getXRate() : BigDecimal.ZERO;
        }
        return BigDecimal.ZERO;
    }

package org.example.service.impl;

import org.example.pojo.McAcFundTxnRecPO;

import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

/**
 * 专门负责 Oracle 侧写入和补偿删除操作。
 * 独立 Bean，确保 @DS("oracle") + @Transactional 通过 Spring AOP 代理生效，
 * 避免同类内部调用导致数据源切换失效的问题。
 */
@Service
@Slf4j
@RequiredArgsConstructor
@DS("oracle")
public class CashVoucherOracleService {

    private static final int ORACLE_REC_SUB_BATCH = 230;
    private static final int ORACLE_SMALL_SUB_BATCH = 1000;

    private final McAcFundRecMapper mcAcFundRecMapper;
    private final McAcFundTxnRecMapper mcAcFundTxnRecMapper;
    private final McFundTpReltnMapper mcFundTpReltnMapper;

    /**
     * Oracle 三张表同一事务写入，原子性保证。
     * 按各表字段数控制子批次，避免 Oracle INSERT ALL 绑定变量超过 32767 限制。
     */
    @Transactional(rollbackFor = Exception.class)
    public void saveAllToOracle(List<McAcFundTxnRecPO> txnList,
                                List<McAcFundRecPO> recList,
                                List<McFundTpReltnPO> tpReltnList) {
        // MC_AC_FUND_TXN_REC: 9字段，每批1000条
        for (int i = 0; i < txnList.size(); i += ORACLE_SMALL_SUB_BATCH) {
            mcAcFundTxnRecMapper.batchInsert(
                    txnList.subList(i, Math.min(i + ORACLE_SMALL_SUB_BATCH, txnList.size())));
        }
        // MC_AC_FUND_REC: 139字段，每批230条（32767/139=235，取230保险）
        for (int i = 0; i < recList.size(); i += ORACLE_REC_SUB_BATCH) {
            mcAcFundRecMapper.batchInsert(
                    recList.subList(i, Math.min(i + ORACLE_REC_SUB_BATCH, recList.size())));
        }
        // MC_FUND_TP_RELTN: 9字段，每批1000条
        for (int i = 0; i < tpReltnList.size(); i += ORACLE_SMALL_SUB_BATCH) {
            mcFundTpReltnMapper.batchInsert(
                    tpReltnList.subList(i, Math.min(i + ORACLE_SMALL_SUB_BATCH, tpReltnList.size())));
        }
    }

    /**
     * 补偿删除：SQL Server 写入失败后删除 Oracle 本批已写数据，保证四表一致。
     * Oracle IN 子句最多 1000 个值（ORA-01795），分批删除。
     */
    @Transactional(rollbackFor = Exception.class)
    public void compensateDeleteOracle(List<Long> txnRidList,
                                       List<Long> fundRidList,
                                       List<String> tpReltnCdeList,
                                       String batchTag) {
        try {
            batchDeleteLong(txnRidList, mcAcFundTxnRecMapper::deleteByIds);
            batchDeleteLong(fundRidList, mcAcFundRecMapper::deleteByIds);
            batchDeleteString(tpReltnCdeList, mcFundTpReltnMapper::deleteByIds);
            log.info("[CashVoucher迁移] batch={} Oracle 补偿删除完成", batchTag);
        } catch (Exception ex) {
            log.error("[CashVoucher迁移] batch={} Oracle 补偿删除失败，需人工介入！txnRids={}",
                    batchTag, txnRidList, ex);
        }
    }

    private void batchDeleteLong(List<Long> ids, java.util.function.Consumer<List<Long>> fn) {
        for (int i = 0; i < ids.size(); i += 1000) {
            fn.accept(ids.subList(i, Math.min(i + 1000, ids.size())));
        }
    }

    private void batchDeleteString(List<String> ids, java.util.function.Consumer<List<String>> fn) {
        for (int i = 0; i < ids.size(); i += 1000) {
            fn.accept(ids.subList(i, Math.min(i + 1000, ids.size())));
        }
    }
}
